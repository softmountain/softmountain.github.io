const int maxn=105;
const int INF=0x3f3f3f3f;
int d[maxn],a[maxn][maxn],v[maxn],ans,m;
void search(){
memset(v,0,sizeof v);
for(int i=1;i<m;i++){
int x=0;
for(int j=1;j<=m;j++){
if(!v[j]&&(x==0||d[j]<d[x])) x=j;
}
v[x]=1;
for(int j=1;j<=m;j++){
d[j]=min(d[j],d[x]+a[x][j]);
}
}
}
